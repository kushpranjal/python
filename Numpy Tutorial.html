"""Deep copy and Shallow copy in numpy"""
"""Deep copy- no link bw the two array"""
"""Shallow copy- when we copy one array to these is a link bw them """
print("Deep Copy")
arr1=np.array([10,20,30])
arr2=arr1.copy()
print("arr1 ->",arr1)
print("arr2 ->",arr2)
arr2[0]=1000
print("after changes in second array\narr1->",arr1,"\narr2->",arr2)
print("\nShallow Copy")
arr3=np.array([10,20,30])
arr4=arr3.view()#view this is like a pointer 
print("arr3 ->",arr3)
print("arr4 ->",arr4)
arr4[0]=1000
print("after making changes arr4[0]=1000")
print("arr3 ->",arr3)
print("arr4 ->",arr4)

Deep Copy
arr1 -> [10 20 30]
arr2 -> [10 20 30]
after changes in second array
arr1-> [10 20 30] 
arr2-> [1000   20   30]

Shallow Copy
arr3 -> [10 20 30]
arr4 -> [10 20 30]
after making changes arr4[0]=1000
arr3 -> [1000   20   30]
arr4 -> [1000   20   30]
